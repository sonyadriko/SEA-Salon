<script>
    document.addEventListener('DOMContentLoaded', function () {
        // Load reviews from the database
        fetch('get_reviews.php')
            .then(response => response.json())
            .then(data => {
                const reviewList = document.getElementById('review-list');
                data.forEach(review => {
                    const reviewItem = document.createElement('div');
                    reviewItem.classList.add('review-item');
                    reviewItem.innerHTML = `
                            <h5>${review.customer_name} - ${review.star_rating} Stars</h5>
                            <p>${review.comment}</p>
                        `;
                    reviewList.appendChild(reviewItem);
                });
            });

        // Load reservations from the database (for demonstration purposes)
        fetch('get_reservations.php')
            .then(response => response.json())
            .then(data => {
                const reservationList = document.getElementById('reservation-list');
                data.forEach(reservation => {
                    const reservationItem = document.createElement('div');
                    reservationItem.classList.add('reservation-item');
                    reservationItem.innerHTML = `
                            <h5>${reservation.name} - ${reservation.service_type}</h5>
                            <p>${reservation.reservation_date} at ${reservation.reservation_time}</p>
                            <p>Phone: ${reservation.phone_number}</p>
                        `;
                    reservationList.appendChild(reservationItem);
                });
            });
    });
</script>
